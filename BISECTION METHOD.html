#include<stdio.h>
#include<math.h>

#define const 0.02
float equ(float x)
{
        return pow(x,3) -5*x + 1;
}

void bisection(float a , float b)
{
        float swap;
        int i;
        float avg;
        float prev;
        if(equ(a)<0  && equ(b)<0 ||equ(a)>0  && equ(b)>0 )
        {
            printf("Both the f(x) are positive or negative.");
            return;
        }
        
            if(equ(a)>0 && equ(b)<0)
        {
            swap = a;
            a  = b;
            b = swap;
            printf("A=%f B=%f",a,b);
        }
        
        for(i=0;;i++)
        {                        
            float arr[20];
            float diff;
            avg = (a+b)/2;
            arr[i]=avg;
            printf("\nIteration %d",i);
            printf("\n\ta = %f b=%f",a,b);
            printf("\n\tAverage  = %f",avg);
            
            if(i>=1)
            {
                if(arr[i]-arr[i-1]>=-0.02 && arr[i]-arr[i-1]<=0.02)
                {
                    printf("\nThe last iteration x = %f \n\t\tdiff = %f",avg,arr[i]-arr[i-1]);
                    return;		
                }
                
            }                               
                    if(equ(avg)<0) 
                {
                        a = avg;
                }
                
                else
                {
                    b = avg;
                }
                    
        }
}
void main()
{
        float a;
        float b;
        printf("Enter the value of a :");
        scanf("%f",&a);
        printf("Enter the value of b :");
        scanf("%f",&b);
        bisection(a,b);
}
